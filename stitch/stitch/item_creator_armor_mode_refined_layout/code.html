using Terraria.ID;
using Terraria.ModLoader;

namespace EtherealWorkbench.Items.Armor
{
    [AutoloadEquip(EquipType.Body)]
    public class ChlorophyteChestplate : ModItem
    {
        public override void SetDefaults() {
            Item.width = 18;
            Item.height = 18;
            Item.scale = 1.0f;
            // Combined: 4 Gold, 80 Silver, 0 Copper
            Item.value = Item.buyPrice(0, 4, 80, 0);
            Item.rare = ItemRarityID.Lime;
            Item.expert = false; // Normal
            Item.defense = 18;
            Item.vanity = false;
        }

        public override void AddRecipes() {
            CreateRecipe()
                .AddIngredient(ItemID.ChlorophyteBar, 24)
                .AddTile(TileID.MythrilAnvil)
                .Register();
        }
    }
}
