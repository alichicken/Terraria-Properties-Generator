using Terraria.ID;
using Terraria.ModLoader;

namespace EtherealWorkbench.Items.Weapons
{
    public class TrueNightsEdge : ModItem
    {
        public override void SetDefaults() {
            Item.damage = 105;
            Item.DamageType = DamageClass.Melee;
            Item.width = 48;
            Item.height = 48;
            Item.useTime = 26;
            Item.useAnimation = 26;
            Item.useStyle = ItemUseStyleID.Swing;
            Item.knockBack = 4.5f;
            Item.value = Item.buyPrice(0, 10, 50, 0);
            Item.rare = ItemRarityID.Yellow;
            Item.autoReuse = true;
            Item.shoot = ModContent.ProjectileType<NightBeam>();
            Item.shootSpeed = 12f;
            Item.mana = 0;
            Item.scale = 1.1f;
        }

        public override void AddRecipes() {
            CreateRecipe()
                .AddIngredient(ItemID.NightsEdge, 1)
                .AddIngredient(ItemID.BrokenHeroSword, 1)
                .AddTile(TileID.MythrilAnvil)
                .Register();
        }
    }
}
